function pick(arr){ return arr[Math.floor(Math.random()*arr.length)] }

function shuffle(arr){
  const a = [...arr];
  for(let i = a.length - 1; i > 0; i--){
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function generateSession(n=10){
  const out = [];
  const types = shuffle(templatePool);
  while(out.length < n){
    const t = templatePool[out.length % templatePool.length];
    out.push(t.build());
  }
  return shuffle(out).slice(0,n);
}

let sessionItems = generateSession(10);
let current = 0;
let score = 0;
let checked = Array(10).fill(false);
let results = Array(10).fill(null);
let started = false;
let finished = false;
let secondsLeft = 300;
let timer = null;
let dragId = null;
let uid = 0;

function createEl(tag, cls, text){
  const e = document.createElement(tag);
  if(cls) e.className = cls;
  if(text !== undefined) e.textContent = text;
  return e;
}

function updateTop(){
  document.getElementById("itemLabel").textContent = `Item ${current + 1} / ${sessionItems.length}`;
  document.getElementById("scoreLabel").textContent = score;
  document.getElementById("timerLabel").textContent = formatTime(secondsLeft);
}

function formatTime(sec){
  const m = String(Math.floor(sec/60)).padStart(2,"0");
  const s = String(sec%60).padStart(2,"0");
  return `${m}:${s}`;
}

function buildTile(text){
  const t = createEl("div","tile",text);
  t.id = "tile_" + (uid++);
  t.draggable = true;
  t.dataset.text = text;
  t.addEventListener("dragstart", e=>{
    dragId = e.target.id;
    e.dataTransfer.setData("text/plain", dragId);
  });
  return t;
}

function allowDrop(e){
  e.preventDefault();
  e.currentTarget.classList.add("over");
}
function leaveDrop(e){
  e.currentTarget.classList.remove("over");
}
function handleDrop(e){
  e.preventDefault();
  e.currentTarget.classList.remove("over");
  const id = e.dataTransfer.getData("text/plain") || dragId;
  const tile = document.getElementById(id);
  if(!tile) return;

  if(e.currentTarget.classList.contains("slot")){
    if(e.currentTarget.children.length === 0){
      e.currentTarget.appendChild(tile);
    }else{
      const existing = e.currentTarget.children[0];
      document.getElementById("bank").appendChild(existing);
      e.currentTarget.appendChild(tile);
    }
  }else{
    e.currentTarget.appendChild(tile);
  }
}

function renderItem(){
  updateTop();
  setStatus("","");
  document.getElementById("review").style.display = "none";
  const item = sessionItems[current];
  document.getElementById("prompt").textContent = item.prompt;

  const sentence = document.getElementById("sentence");
  sentence.innerHTML = "";

  const slotsWrap = createEl("span","slots");
  item.chunks.forEach((_,i)=>{
    const slot = createEl("span","slot");
    slot.dataset.index = i;
    slot.addEventListener("dragover",allowDrop);
    slot.addEventListener("dragleave",leaveDrop);
    slot.addEventListener("drop",handleDrop);
    slotsWrap.appendChild(slot);
  });

  sentence.appendChild(slotsWrap);

  const bank = document.getElementById("bank");
  bank.innerHTML = "";
  bank.addEventListener("dragover",allowDrop);
  bank.addEventListener("dragleave",leaveDrop);
  bank.addEventListener("drop",handleDrop);

  shuffle(item.chunks).forEach(ch => bank.appendChild(buildTile(ch)));
}

function getUserChunks(){
  return [...document.querySelectorAll(".slot")].map(s => s.firstElementChild ? s.firstElementChild.dataset.text : "");
}

function normalize(s){
  return s.replace(/\s+/g," ").trim();
}

function checkAnswer(){
  if(!started || finished){
    setStatus("Press Start to begin the timed set.","warn");
    return;
  }
  const user = getUserChunks();
  if(user.includes("")){
    setStatus("Fill every slot before checking.","warn");
    return;
  }

  const userSentence = normalize(user.join(" ")) + (sessionItems[current].answer.endsWith("?") ? "?" : ".");
  const correct = sessionItems[current].answer;
  const ok = userSentence === correct;

  if(!checked[current]){
    checked[current] = true;
    if(ok) score++;
  }else{
    const wasCorrect = results[current] === true;
    if(wasCorrect && !ok) score--;
    if(!wasCorrect && ok) score++;
  }
  results[current] = ok;
  updateTop();

  if(ok){
    setStatus("Correct.","ok");
  }else{
    setStatus("Not correct yet. Recheck question order or embedded-clause order.","bad");
  }
}

function resetItem(){
  if(finished) return;
  renderItem();
}

function nextItem(){
  if(finished) return;
  current = (current + 1) % sessionItems.length;
  renderItem();
}

function startQuiz(){
  if(started) return;
  started = true;
  timer = setInterval(()=>{
    secondsLeft--;
    updateTop();
    if(secondsLeft <= 0){
      finishQuiz();
    }
  },1000);
  setStatus("Timer started.","ok");
}

function finishQuiz(){
  if(finished) return;
  finished = true;
  clearInterval(timer);
  updateTop();

  const review = document.getElementById("review");
  const list = document.getElementById("reviewList");
  list.innerHTML = "";

  sessionItems.forEach((item,i)=>{
    const row = createEl("div","review-item");
    row.innerHTML = `
      <div><strong>Item ${i+1}:</strong> ${item.prompt}</div>
      <div><strong>Status:</strong> ${results[i] === true ? "Correct" : "Not correct / not checked"}</div>
      <div><strong>Answer:</strong> ${item.answer}</div>
    `;
    list.appendChild(row);
  });
  review.style.display = "block";
  setStatus(`Finished. Final score: ${score} / ${sessionItems.length}`,"ok");
}

function newSet(){
  clearInterval(timer);
  sessionItems = generateSession(10);
  current = 0;
  score = 0;
  checked = Array(10).fill(false);
  results = Array(10).fill(null);
  started = false;
  finished = false;
  secondsLeft = 300;
  updateTop();
  renderItem();
  setStatus("New random set generated. Press Start.","ok");
}

function setStatus(text,type){
  const box = document.getElementById("status");
  box.className = "status";
  box.textContent = text;
  if(type) box.classList.add(type);
}

document.getElementById("startBtn").addEventListener("click", startQuiz);
renderItem();
